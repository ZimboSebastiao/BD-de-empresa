CREATE DATABASE FUNCIONARIOS;

CREATE TABLE DEPARTAMENTOS
(
CODIGO INT(3) PRIMARY KEY,
NOME VARCHAR(70),
LOCALIZACAO varchar(30)
);

CREATE TABLE DEPENDENTES 
(
CODIGO INT(4) PRIMARY KEY,
NOME VARCHAR(70),
ENDERECO VARCHAR(50),
PARENTESCO VARCHAR(10)
);

CREATE TABLE FUNCIONARIOS
(
CPF NUMERIC(15) PRIMARY KEY,
NOME VARCHAR(70),
ENDERECO VARCHAR(30),
TELEFONE NUMERIC(10),
SALARIO DECIMAL(6,2),
CODIGO_DEPARTAMENTO INT(3) REFERENCES DEPARTAMENTOS,
CODIGO_DEPARTAMENTOGERENCIAR INT(3) REFERENCES DEPARTAMENTOS
);

CREATE TABLE FUNCIONARIO_DEPENDENTES
(
CPF_FUNCIONARIO NUMERIC(15) REFERENCES FUNCIONARIOS,
CODIGO_DEPENDENTE INT(4) REFERENCES DEPENDENTES
);

CREATE TABLE PROJETOS
(
NUMERO INT(2) PRIMARY KEY,
NOME VARCHAR(70),
TIPO VARCHAR(30),
VERBA VARCHAR(35),
CODIGO_DEPARTAMENTO INT(3) REFERENCES DEPARTAMENTOS,
CPF_FUNCIONARIO NUMERIC(15) REFERENCES FUNCIONARIOS
);

CREATE TABLE FUNCIONARIO_PROJETOS
(
CPF_FUNCIONARIO NUMERIC(15) REFERENCES FUNCIONARIOS,
NUMERO_PROJETO INT(2) REFERENCES PROJETOS,
NUMERO_HORAS INT(4)
);
